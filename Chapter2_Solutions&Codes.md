## 2.1-2)
========
```
#include <stdio.h>
#include <stdlib.h>

int main()
{
    int a[] = {31, 41, 59, 26, 41, 58};
    for(int i=1;i<6;i++){
        int j = i-1, key = a[i];
        while(j >= 0 && a[j] < key){
            a[j+1] = a[j];
            j--;
        }
        a[j+1] = key;
    }

    for(int i=0;i<6;i++){
        printf("%i\t",a[i]);
    }
    return 0;
}
```
